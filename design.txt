      tryMeBtn.addEventListener('click', function() {

          initialContent.style.display = 'none';

      function animateElements() {
        document.querySelectorAll('.fade-in-element').forEach((el, index) => {
          setTimeout(() => el.style.opacity = '1', 300 * (index + 1));
        });
        setTimeout(() => {
          searchBar.style.opacity = '1';
          searchButton.style.opacity = '1';
        }, 600);
      }

      // 3. FuncÈ›ie universalÄƒ pentru cÄƒutare
      function handleSearch(inputElement, isCompact = false) {
        const gene = inputElement.value.trim();
        if (!gene) return;

        // Fade-out la conÈ›inutul curent

          fadeInContent.style.opacity = '0';
          setTimeout(() => fadeInContent.style.display = 'none', 800);
        }

        // Fade-in pentru bara compactÄƒ
        compactSearchBar.value = gene;
        if (compactSearchContainer.style.display !== 'block') {
          compactSearchContainer.style.display = 'block';
          setTimeout(() => compactSearchContainer.style.opacity = '1', 50);
        }

        // Aici poÈ›i adÄƒuga logica pentru afiÈ™area rezultatelor

      // 4. Evenimente de cÄƒutare

      searchBar.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') handleSearch(searchBar);
      });

      compactSearchBtn.addEventListener('click', () => handleSearch(compactSearchBar, true));
      compactSearchBar.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') handleSearch(compactSearchBar, true);
      });

document.addEventListener('DOMContentLoaded', function() {
  // Elemente DOM
  const tryMeBtn = document.getElementById('tryMeBtn');
  const initialContent = document.getElementById('initial-content');
  const fadeInContent = document.getElementById('fade-in-content');
  const searchBar = document.getElementById('search-bar');
  const searchButton = document.querySelector('.search-button');
  const compactSearchContainer = document.getElementById('compact-search-container');
  const compactSearchBar = document.getElementById('compact-search-bar');
  const resetSearchBtn = document.getElementById('reset-search');

  // 1. Ascunde conÈ›inutul iniÈ›ial È™i afiÈ™eazÄƒ bara de cÄƒutare
  tryMeBtn.addEventListener('click', function() {
    initialContent.style.opacity = '0';
    setTimeout(() => {
      initialContent.style.display = 'none';
      fadeInContent.style.display = 'flex';
      
      // AnimaÈ›ie fade-in pentru elemente
      document.querySelectorAll('.fade-in-element').forEach((el, index) => {
        setTimeout(() => {
          el.style.opacity = '1';
        }, 300 * (index + 1));
      });
      
      setTimeout(() => {
        searchBar.style.opacity = '1';
        document.querySelector('.search-button').style.opacity = '1';
      }, 600);
    }, 550);
  });

  // 2. La cÄƒutare, mutÄƒ bara Ã®n partea de sus
  searchButton.addEventListener('click', handleSearch);
  searchBar.addEventListener('keypress', function(e) {
    if (e.key === 'Enter') handleSearch();
  });

  function handleSearch() {
    const gene = searchBar.value.trim();
    if (!gene) return;

    // Fade-out la conÈ›inutul curent
    fadeInContent.style.opacity = '0';
    
    setTimeout(() => {
      fadeInContent.style.display = 'none';
      
      // SeteazÄƒ valoarea Ã®n bara compactÄƒ
      compactSearchBar.value = gene;
      compactSearchContainer.style.display = 'block';
      
      // Fade-in pentru bara compactÄƒ
      setTimeout(() => {
        compactSearchContainer.style.opacity = '1';
        
        // Aici poÈ›i adÄƒuga logica pentru afiÈ™area rezultatelor
        console.log(`Searching for: ${gene}`);
      }, 50);
    }, 800);
  }

  // 3. Resetare la starea iniÈ›ialÄƒ
  resetSearchBtn.addEventListener('click', function() {
    compactSearchContainer.style.opacity = '0';
    setTimeout(() => {
      compactSearchContainer.style.display = 'none';
      initialContent.style.display = 'flex';
      initialContent.style.opacity = '1';
      searchBar.value = '';
    }, 800);
  });
});
